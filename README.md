🎬 What I Do with Manim
from manim import *

class WhoIsShravan(Scene):
    def construct(self):
        title = Text("Shravan Kumar", font_size=56, color=BLUE)
        role  = Text("Student · Coder · Animator", font_size=28, color=YELLOW)
        role.next_to(title, DOWN)

        self.play(Write(title), run_time=2)
        self.play(FadeIn(role, shift=UP))
        self.wait(2)

        quote = Text("\"Animate the world with code\"", font_size=22, color=WHITE, slant=ITALIC)
        self.play(Transform(title, quote))
        self.wait(2)
